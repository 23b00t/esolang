def initialize(code, iterations, width, height)

def run
  while running? do
    case command
    when 'n' then move_up
    when 'e' then move_right
    when 's' then move_down
    when 'w' then move_left
    when '*' then flip
    when '[' then loop_begin
    when ']' then loop_end
    end

    @code_pointer += 1
    @iterations -= 1
  end

  output
end
